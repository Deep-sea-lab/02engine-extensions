使用方法

1. 引入库

<script src="神经网络扩展2.0.js"></script>

2. 创建神经网络

// 创建神经网络实例 const neuralNetwork = new NeuralNetwork(); // 配置网络结构 neuralNetwork.addLayer(2); // 输入层：2个神经元 neuralNetwork.addLayer(4); // 隐藏层：4个神经元 neuralNetwork.addLayer(1); // 输出层：1个神经元 // 设置学习率 neuralNetwork.setLearningRate(0.1);

3. 训练网络

// 准备训练数据 const trainingData = [ { inputs: [0, 0], targets: [0] }, { inputs: [0, 1], targets: [1] }, { inputs: [1, 0], targets: [1] }, { inputs: [1, 1], targets: [0] } ]; // 训练网络 for (let epoch = 0; epoch < 10000; epoch++) { // 随机选择一个训练样本 const data = trainingData[Math.floor(Math.random() * trainingData.length)]; // 训练网络 const result = neuralNetwork.train(data.inputs, data.targets); // 每1000次迭代打印一次进度 if (epoch % 1000 === 0) { console.log(`迭代次数: ${epoch}, 结果: ${result}`); } } console.log('训练完成！');

4. 使用网络进行预测

// 使用训练好的网络进行预测 console.log('测试输入 [0, 0]:', neuralNetwork.feedForward([0, 0])); console.log('测试输入 [0, 1]:', neuralNetwork.feedForward([0, 1])); console.log('测试输入 [1, 0]:', neuralNetwork.feedForward([1, 0])); console.log('测试输入 [1, 1]:', neuralNetwork.feedForward([1, 1]));

5. 启用可视化

// 创建可视化面板 createVisualizationPanel(); // 绘制网络结构 drawNetworkStructure(neuralNetwork);
